#  1、c语言实现 
    
```
#include<stdio.h>

typedef  enum {
	SMALL,
	MEDUIM,
	LARGE
}JuiceEnum;

typedef struct {
	JuiceEnum juicesize;
}JuiceStru;

int main()
{
	JuiceStru juice = {
		SMALL,
	};
	
	printf("juice is %d\n", juice.juicesize);
	juice.juicesize = LARGE;
	printf("juice is %d\n", juice.juicesize);

	return 0;
}
```
